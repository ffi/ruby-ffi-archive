module HelloWin32
  extend FFI::Library

  ffi_lib :user32
  ffi_convention :stdcall

  attach_function :message_box, :MessageBoxA,
[ :pointer, :string, :string, :uint ], :int
end

rc = HelloWin32.message_box nil, 'Hello Win32 World!', 'FFI Win32
Example', 1
puts "Return code: #{rc}"
